BUS RAM
	RAM Instruction: Multiple at a time
		0b000001:jmp [data]
		0b000010:mov ip,[ip]
		0b000100:store at IP
		0b001000:Read To Bus at IP
		0b010000:Read/Write BUS
		0b100000:Inc address pointer
	RAM Register: One at a time
		0b000:mov reg,value16
		0b001:mov regH,value8
		0b010:mov regL,value8
		0b011:dec reg
		0b100:inc reg
		0b101:set 0
		0b110:Read/Write BUS
